// The asteroid behaves differently based on whether it has a parent or not. Without one it'll
// just move off in a given direction (passed_param_1 = angle) but with one it'll accellerate
// towards said planet so it crashes into the interior of it it and explodes, leaving a mark if it hits the front.

#define planet_section 		passed_param_1
#define main_movement_angle	passed_param_2
#define angle_offset		passed_param_3

#define movement_speed		v3
#define actual_speed		v4
#define generation		v5
#define total_lifetime		v6
#define sub_child		v7

#define x_offset		v9
#define y_offset		v10

#define planet_type		v29

#define start_speed		500

	let total_lifetime = rand 120 360

	let actual_speed = rand 0 start_speed

	let x_vel = actual_speed cos main_movement_angle
	let y_vel = actual_speed sin main_movement_angle

	let lifetime = 30

	let entity_type = entity_type + ENT_TYPE_DISPOSABLE
	let entity_type = entity_type + ENT_TYPE_SLEEPABLE

	let opengl_booleans = opengl_booleans + opengl_boolean_scale
	let opengl_booleans = opengl_booleans + opengl_boolean_rotate
	let opengl_booleans = opengl_booleans + opengl_boolean_filtered

	let opengl_scale_x = rand 7000 10000
	let opengl_scale_y = opengl_scale_x
	let temp_1 = rand 5000 15000
	let opengl_scale_y = opengl_scale_y % temp_1

	let opengl_scale_x = opengl_scale_x % parent.opengl_scale_x
	let opengl_scale_y = opengl_scale_y % parent.opengl_scale_x

	let Bitshift = 8
	SET_PRIVATE_CO_ORDS
	let sprite = planet_masks[arb]

	if sub_child = false then
		let base_frame = parent.planet_type	// So the texture is right
		let base_frame = base_frame * 2
		let temp_1 = rand 0 1			// So it's randomly the big or small one
		let base_frame = base_frame + temp_1
		let base_frame = base_frame + 9		// So it's not pointing at the planet masks
	else
		let base_frame = parent.base_frame
	end_if

	let current_frame = base_frame

	let draw_mode = draw_mode_sprite

	let draw_order = 2 // Behind the planet chunks

	let angle_speed = rand -100 100


start

	let opengl_angle = opengl_angle + angle_speed

	if lifetime > 0 then
		let lifetime = lifetime - 1
		let x_vel = x_vel % 9500
		let y_vel = y_vel % 9500
	end_if

	if total_lifetime > 0 then
		let total_lifetime = total_lifetime - 1
	else
		if generation > 0 then
			for temp_1 = 0 to 2
				let x_offset = rand -6 6
				let y_offset = rand -6 6
				spawn_entity ex_planet_shatter_asteroid x_offset y_offset process_next
				let last_child.generation = generation - 1
				let last_child.sub_child = true
				let last_child.main_movement_angle = rand 0 36000
			next
		end_if

		spawn_entity ex_player_rocket_explosion_flame 0 0 process_next

		let last_child.opengl_angle = rand 0 36000
		let last_child.opengl_scale_x = opengl_scale_x / 2
		let last_child.opengl_scale_y = last_child.opengl_scale_x
		let last_child.draw_order = last_child.draw_order + 1

		kill_entity own_id
	end_if

end
