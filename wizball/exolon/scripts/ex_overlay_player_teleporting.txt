#define adder v0
#define twinkle_x_offset v1
#define twinkle_y_offset v2
#define twinker_counter v3
#define overlay_id v4

let entity_type = entity_type + ENT_TYPE_SLEEPABLE

spawn_entity ex_standard_overlay 0 0 PROCESS_NEXT
let overlay_id = last_child
let overlay_id.secondary_frame = 4
let overlay_id.passed_param_1 = parent
let last_child.passed_param_2 = opengl_boolean_blend_multiply

let adder = 15
let counter = 0
let signal = 0

start
	let overlay_id.opengl_vertex_alpha = counter

	let counter = counter + adder

	if counter > 255 then
		let adder = -15

		let world_x = parent.world_x
		let world_y = parent.world_y - 32

		set_private_co_ords

		spawn_entity ex_player_teleport_mask 0 0 PROCESS_NEXT
		let last_child.facing_direction = parent.facing_direction

		for twinker_counter = 0 to 20
			let twinkle_x_offset = rand -12 12
			let twinkle_y_offset = rand -32 32
			spawn_entity ex_twinkle twinkle_x_offset twinkle_y_offset PROCESS_NEXT
		next

		let signal = 1
	end_if

	if counter < 0 then
		let overlay_id.signal = 1
		let signal = 2
		kill_entity own_id
	end_if
end
