#define fade_stage		v0

#define fade_up			0
#define fade_stay		1
#define fade_down		2

	let entity_type = entity_type + ENT_TYPE_DISPOSABLE

	let Bitshift = 0
	SET_PRIVATE_CO_ORDS

	let draw_order = 5

	let sprite = large_graphics[arb]
	let base_frame = 1
	let current_frame = base_frame
	let draw_mode = DRAW_MODE_SPRITE
	let opengl_booleans = OPENGL_BOOLEAN_BLEND_ADD
	let opengl_booleans = opengl_booleans + OPENGL_BOOLEAN_VERTEX_COLOUR

	let world_x = 320
	let world_y = 240

	set_private_co_ords

	let opengl_vertex_red = 0
	let opengl_vertex_green = 0
	let opengl_vertex_blue = 0

	let fade_stage = fade_up

	let counter = 0
	
start
	if fade_stage = fade_up then

		if opengl_vertex_blue < 128 then
			let opengl_vertex_blue = opengl_vertex_blue + 3
		else_if opengl_vertex_green < 255 then
			if opengl_vertex_blue < 255 then
				let opengl_vertex_blue = opengl_vertex_blue + 3
			end_if
			let opengl_vertex_green = opengl_vertex_green + 3
			let opengl_vertex_red = opengl_vertex_green
			if opengl_vertex_red = 255 then
				let fade_stage = fade_stay 
			end_if
		end_if

	else_if fade_stage = fade_stay then

		let counter = counter + 1
		if counter = 120 then
			let counter = 0
			let fade_stage = fade_down
		end_if

	else_if fade_stage = fade_down then

		if opengl_vertex_red > 0 then
			let opengl_vertex_red = opengl_vertex_red - 3
			let opengl_vertex_green = opengl_vertex_red
			let opengl_vertex_blue = opengl_vertex_red
		else
			kill_entity own_id
		end_if

	end_if

end
