#define offset_x		v2
#define offset_y		v3
#define distance		v4
#define offset_angle		v5

#define behaviour		v0
#define fade_up			0
#define stable			1
#define fade_down		2

	let entity_type = entity_type + ENT_TYPE_DISPOSABLE

	let Bitshift = 4
	SET_PRIVATE_CO_ORDS

	let draw_mode = DRAW_MODE_SPRITE
	let COLLISION_SHAPE = RECTANGLE

	let opengl_booleans = opengl_boolean_blend_multiply + opengl_boolean_vertex_colour_alpha

	let sprite = level_scenery_sprites[arb] // Handy place to shove 'em. Doh!

	let opengl_vertex_alpha = 0

	let temp_1 = rand 60

	start
		if parent.signal & destroy_main_text then
			kill_entity own_id
		end_if

		let temp_1 = temp_1 - 1

		if temp_1 <= 0 then
			goto continue
		end_if
	end

.continue

	let base_x = x
	let base_y = y

	let offset_angle = rand 0 36000
	let distance = 256

	let behaviour = fade_up

start

	if distance > 0 then
		let distance = distance - 8

		let offset_x = distance sin offset_angle
		let offset_y = distance COS offset_angle

		let offset_angle = offset_angle + 500

		let x = base_x + offset_x
		let y = base_y + offset_y
	end_if

	if behaviour = fade_up then
		if opengl_vertex_alpha < 255 then
			let opengl_vertex_alpha = opengl_vertex_alpha + 15
		else
			let opengl_booleans = opengl_boolean_masked
			let behaviour = stable
		end_if
	else_if behaviour = fade_down then
		if opengl_vertex_alpha > 0 then
			let opengl_vertex_alpha = opengl_vertex_alpha - 15
		else
			kill_entity own_id
		end_if
	end_if

	if parent.signal & destroy_main_text then
		let opengl_booleans = opengl_boolean_blend_multiply + opengl_boolean_vertex_colour_alpha
		let behaviour = fade_down
	end_if

end
