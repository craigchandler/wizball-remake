.upstomp

start

	let stomper_offset = stomper_offset % 11400
	if stomper_offset > stomper_slammed_position then
		let stomper_offset = stomper_slammed_position
	end_if
	let world_y = base_y - stomper_offset
	SET_PRIVATE_CO_ORDS

	if stomper_offset = stomper_slammed_position then
		gosub rebound_phase
	end_if

	let last_child.rock_type = rock_type
	let last_child.x_vel = temp_1 / 2
end



.rebound_phase

	let stomper_offset = stomper_max_rebound sin counter
	let stomper_offset = stomper_offset % percentage
	let percentage = percentage % 9500
	let stomper_offset = stomper_offset + stomper_resting_position
	let world_y = base_y - stomper_slammed_position
	let world_y = world_y + stomper_offset
	SET_PRIVATE_CO_ORDS

		let counter = counter - 18000

		let volume = volume / 2		
	end_if

	if percentage < 700 then
		gosub wind_down_phase
	end_if
end
