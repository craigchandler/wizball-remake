// These are the red arrows that pulse inwards towards the score until it's selected.

#define left_or_right		passed_param_1
#define left			0
#define right			1
#define start_x			v0

let sprite = gameover[arb]

let bitshift = 0

if left_or_right = left then
	let current_frame = 10
	let x_vel = -1
	let start_x = parent.world_x + 80
else
	let current_frame = 9
	let x_vel = 1
	let start_x = parent.world_x - 80
end_if

let world_x = start_x

let opengl_booleans = opengl_booleans + opengl_boolean_vertex_colour

set_private_co_ords

let opengl_vertex_red = 255
let opengl_vertex_green = 255
let opengl_vertex_blue = 255

let draw_order = 21

let draw_mode = draw_mode_sprite

start
	let opengl_vertex_red = opengl_vertex_red - 5
	let opengl_vertex_green = opengl_vertex_red
	let opengl_vertex_blue = opengl_vertex_red

	if parent.signal != CONST_UNSET then
		let signal = 1
	end_if

	if opengl_vertex_red < 0 then
		if signal = 1 then
			kill_entity own_id
		else
			let world_x = start_x

			let opengl_vertex_red = 255
			let opengl_vertex_green = 255
			let opengl_vertex_blue = 255

			set_private_co_ords
		end_if
	end_if
end
