#define animation_speed		2
#define end_frame		v0
#define opening			2
#define destructing		1
#define closing			0

	let sprite = large_graphics[arb]
	let base_frame = 17
	let end_frame = base_frame + 6
	let current_frame = base_frame
	let draw_mode = draw_mode_sprite

	if parent.signal = closing then // shut
		let current_frame = base_frame
	end_if

	if parent.signal = opening then // open
		let current_frame = end_frame
	end_if

	let draw_order = draw_order + 2

start
	let counter = counter + 1 !> animation_speed ? 0

	let world_x = parent.world_x
	let world_y = parent.world_y

	if counter = 0 then
		if parent.signal = closing then
			if draw_mode = draw_mode_invisible then
				let draw_mode = draw_mode_sprite
			else
				let current_frame = current_frame - 1 !< base_frame
			end_if
		else_if parent.signal = opening then
			if current_frame = end_frame then
				let draw_mode = draw_mode_invisible
			end_if
			let current_frame = current_frame + 1 !> end_frame
		end_if
	end_if

	if parent.signal = destructing then
		kill_entity own_id
	end_if
end
