// ---------------- Object Interaction ----------------

.object_interaction_routine

	IF collided_entity.COLLIDE_TYPE & ENT_TYPE_CATELLITE then

		IF special_bonus_flag = CONST_UNSET THEN

			let temp_1 = get_global_flag (MAIN_GAME_CONTROLLER_ENTITY_ID)

			if temp_1 != CONST_UNSET THEN
				LET temp_2 = ADD_PAINTDROP_RED + colour_flag
				// This now equals the right signal for the given colour.

				ADD_EVENT_TO_QUEUE ( temp_1 , temp_2 )

				PLAY_SOUND paintdrop_collection 255 CONST_SOUND_FREQUENCY
			ELSE
				STOP // WTF?!
			END_IF

		ELSE

			IF_FLAG (wizball_entity_id_flag != CONST_UNSET) then

				switch special_bonus_flag

					case PAINTDROP_FILTH_RAID
						PLAY_SOUND special_paintball_pickup_filth_raid 255 CONST_SOUND_FREQUENCY
						spawn_entity (spawn_fuzz,0,0,process_next)
						let last_child.passed_param_1 = 3
					break

					case PAINTDROP_INDESTRUCTACAT
						PLAY_SOUND special_paintball_pickup_indestructacat 255 CONST_SOUND_FREQUENCY
						set_global_flag (catellite_stored_health,128)
					break

					case PAINTDROP_MUTANT_CAT
						PLAY_SOUND special_paintball_pickup_mutant_cat 255 CONST_SOUND_FREQUENCY
						set_global_flag (mutant_cat_flag,TRUE)
					break

					case PAINTDROP_BONUS_WIZBALL
						PLAY_SOUND special_paintball_pickup_extra_life 255 CONST_SOUND_FREQUENCY
						call_script function_gain_life
					break

					case PAINTDROP_FREAKY_BITS
						PLAY_SOUND special_paintball_pickup_freaky_bits 255 CONST_SOUND_FREQUENCY
						spawn_entity (freakout_effect_manager,0,0,process_next)
					break

				end_switch

			END_IF

		END_IF

		let particle_count = get_global_flag (particle_commonness_multiplier)
		let particle_count = 3 * particle_count

		for counter = 1 to particle_count 
			let temp_1 = rand (-8,8)
			let temp_2 = rand (-16,16)

			spawn_entity particle_twinkle,temp_1,temp_2,process_next
		next

		kill_entity own_id
	END_IF

end

// ---------------- World Interaction ----------------

.world_interaction_routine

	if WORLD_HIT_AXIS = VERTICAL_AXIS then

		PLAY_SOUND paintdrop_splat 255 CONST_SOUND_FREQUENCY

		spawn_entity (paintdrop_stain,0,12,process_next)
		LET last_child.colour_flag = colour_flag
		LET last_child.special_bonus_flag = special_bonus_flag

		let angular_step = get_global_flag (particle_rareness_multiplier)
		let angular_step = 2000 * angular_step

		for counter = 9000 to 27000 step angular_step
			spawn_entity (particle_small_colour_drip,0,0,process_next)
			LET last_child.colour_flag = colour_flag
			LET last_child.special_bonus_flag = special_bonus_flag
			LET temp_2 = rand -800,800
			LET last_child.burst_angle = counter + temp_2
		next

		kill_entity own_id

	end_if

end
