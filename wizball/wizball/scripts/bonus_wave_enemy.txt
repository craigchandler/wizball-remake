.main_loop

start

	if debugging_this_type = true then
		start_program_trace
	end_if

	fixed_switch enemy_type

		fixed_case BONUS_WAVE_TYPE_SLOW_PLANES
			let behaviour_counter = 36000 * world_x
			let behaviour_counter = behaviour_counter / wave_length

			if (wave_flags & Y_INVERT) then
				let world_y = STARTING_WORLD_Y - wave_height
				let temp_1 = wave_height cos behaviour_counter
				let world_y = world_y - temp_1

				let current_frame = frame_count cos behaviour_counter
				let current_frame = base_frame - current_frame 
			else
				let world_y = STARTING_WORLD_Y + wave_height
				let temp_1 = wave_height cos behaviour_counter
				let world_y = world_y + temp_1

				let current_frame = frame_count cos behaviour_counter
				let current_frame = base_frame - current_frame 
			end_if

			set_private_y_co_ord

			let velocity_counter = velocity_counter + 360 !> 0

			set_global_flag (bonus_debug_flag,x_vel)

			let x_vel = 512 cos velocity_counter
			let x_vel = x_vel + 512
		break

		fixed_case BONUS_WAVE_TYPE_REGULAR_PAINTBALL_BOUNCE
			let deformation_angle = deformation_angle + 1000 !< 0 !> 35999
			let deformation_amount = deformation_amount - 100 !< 0

			let temp_1 = deformation_amount cos deformation_angle

			let opengl_scale_x = 10000 + temp_1
			let opengl_scale_y = 10000 - temp_1
		break

		fixed_case BONUS_WAVE_TYPE_RANDOM_ASTEROIDS
			let current_frame = current_frame + 1
			set_anim_frame (ANIM_LOOP, base_frame, frame_count, 4, animation_counter, false)

			let animation_counter = animation_counter + 1

			let opengl_angle = opengl_angle - rotation_speed
		break

		fixed_case BONUS_WAVE_TYPE_RANDOM_CIRCLES
			let behaviour_counter = behaviour_counter + 1

			if behaviour_counter = circle_acceleration_timer then
				let x_acc = 0
				let y_acc = 0
			end_if

			let opengl_vertex_red = 192 sin animation_counter
			let opengl_vertex_red = opengl_vertex_red + 64
			let opengl_vertex_green = opengl_vertex_red + 256

			// Just make sure it ain't really -128...
			if opengl_vertex_red < 0 then
				let opengl_vertex_red = 0
			end_if

			let animation_counter = animation_counter + 750 !< 0 !> 35999
		break

		fixed_case BONUS_WAVE_TYPE_RANDOM_PAINTBALL_BOUNCE
			let opengl_angle = opengl_angle - rotation_speed
		break

		fixed_case BONUS_WAVE_TYPE_FILTH
			if PATH_CURRENT_PERCENTAGE < 1000000 then
				LET PATH_CURRENT_OFFSET_X, PATH_CURRENT_OFFSET_Y, PATH_CURRENT_SECTION = GET_SPECIAL_PATH_POSITION_TOTAL_OFFSET (PATH_NUMBER, PATH_ANGLE, PATH_CURRENT_PERCENTAGE, PATH_CURRENT_SECTION)

				LET PATH_CURRENT_PERCENTAGE = PATH_CURRENT_PERCENTAGE + PATH_PERCENTAGE_SPEED

				if (wave_flags & Y_INVERT) then
					LET WORLD_Y = STARTING_WORLD_Y - PATH_CURRENT_OFFSET_Y
				else
					LET WORLD_Y = STARTING_WORLD_Y + PATH_CURRENT_OFFSET_Y
				end_if

				if (wave_flags & X_INVERT) then
					LET WORLD_X = STARTING_WORLD_X - PATH_CURRENT_OFFSET_X
				else
					LET WORLD_X = STARTING_WORLD_X + PATH_CURRENT_OFFSET_X
				end_if

				SET_PRIVATE_CO_ORDS
			end_if

			set_anim_frame (ANIM_LOOP, base_frame, frame_count, 2, animation_counter, false)

			let animation_counter = animation_counter + 1
		break

		fixed_case BONUS_WAVE_TYPE_BONUS_LIFE
			let current_frame = current_frame + 1 !< 0 !> 63
		break

		fixed_case BONUS_WAVE_TYPE_NEW_8_WAY_SHOOTERS
			set_anim_frame (ANIM_LOOP, base_frame, frame_count, 4, animation_counter, false)
			let animation_counter = animation_counter + 1

			if (wave_flags & ALTERNATE_MODE_1) then
				// Alternate 1 - appears, moves a bit vertically, stops, shoots, etc.

				LET OPENGL_VERTEX_ALPHA = OPENGL_VERTEX_ALPHA + 15 !> 255

				let behaviour_counter = behaviour_counter + 1
				if behaviour_counter = 90 then
					let x_vel = stored_x_vel
					let y_vel = stored_y_vel
				else_if behaviour_counter = 480 then
					let stored_x_vel = x_vel
					let stored_y_vel = y_vel
					let x_vel = 0
					let y_vel = 0
					let behaviour_counter = 0
				end_if

			end_if
		break

		fixed_case BONUS_WAVE_TYPE_NEW_ROTATE_SHOOTERS
			set_anim_frame (ANIM_LOOP, base_frame, frame_count, 4, animation_counter, false)
			let animation_counter = animation_counter + 1

			LET OPENGL_VERTEX_ALPHA = OPENGL_VERTEX_ALPHA + 15 !> 255
		break

		fixed_case BONUS_WAVE_TYPE_UP_AND_DOWNERS
			let temp_1 = 64 sin sine_wave_counter 

			let world_y = STARTING_WORLD_Y - temp_1

			SET_PRIVATE_Y_CO_ORD

			let sine_wave_counter = sine_wave_counter + 400

			set_anim_frame (ANIM_PING_PONG, base_frame, frame_count, 2, animation_counter, false)

			let animation_counter = animation_counter + 1
		break


		fixed_case BONUS_WAVE_TYPE_NEW_RANDOM_WAY_DRIVE_BYS

		break

	end_switch

	gosub fire_shots

	gosub check_if_off_screen_and_retire

end

// ---------------- Object Interaction ----------------

.object_interaction_routine

	// Blow 'em up real good! Unless they're an asteroid.

	if enemy_type = BONUS_WAVE_TYPE_RANDOM_ASTEROIDS then
		// If an asteroid hits a shield, it blows up to avoid depleting it like mad...

		if collided_entity.entity_type & ENT_TYPE_SHIELD_ENTITY then
			if wave_continue_mode = WAIT_UNTIL_LAST_DEAD then
				let parent.total_alive_children = parent.total_alive_children - 1
			end_if

			call_script function_kill_normal_enemy
		else
			// Otherwise, if they're heading left, then reverse their direction.
			if x_vel < 0 then
				
				PLAY_SOUND asteroid_scrape 255 CONST_SOUND_FREQUENCY

				let x_vel = -x_vel
				let rotation_speed = -rotation_speed
			end_if
		end_if
	else
		// Otherwise... KABOOM!

		if wave_continue_mode = WAIT_UNTIL_LAST_DEAD then
			let parent.total_alive_children = parent.total_alive_children - 1
		end_if

		if enemy_type = BONUS_WAVE_TYPE_BONUS_LIFE then
			call_script function_gain_life
		end_if

		let temp_1 = get_global_flag (bonus_level_enemies_killed)
		let temp_1 = temp_1 + 1
		set_global_flag (bonus_level_enemies_killed, temp_1)

		let temp_1 = wave_number_in_bonus_level / 3
		let temp_1 = temp_1 * 10
		let temp_1 = temp_1 + 20

		let temp_2 = get_global_flag (player_score)
		let temp_2 = temp_2 + temp_1 !> MAXIMUM_POSSIBLE_SCORE
		set_global_flag (player_score,temp_2)

		call_script function_kill_normal_enemy
	end_if
end

// ---------------- World Interaction ----------------

.world_interaction_routine

	// Actually, we don't really need one. We can let the autonomous systems take care of everything I reckon.

	// Well, apart from the bubbles. D'oh!

	PLAY_SOUND enemy_bounce 64 CONST_SOUND_FREQUENCY

	if enemy_type = BONUS_WAVE_TYPE_REGULAR_PAINTBALL_BOUNCE then
		let deformation_angle = 0
		let deformation_amount = 6000
	else_if enemy_type = BONUS_WAVE_TYPE_RANDOM_PAINTBALL_BOUNCE then
		let deformation_angle = 0
		let deformation_amount = 6000
	end_if

end

// ---------------- Termination ----------------

.check_if_off_screen_and_retire

	if world_x < LEFT_SIDE_BASE_X then
		if wave_continue_mode = WAIT_UNTIL_LAST_DEAD then
			let parent.total_alive_children = parent.total_alive_children - 1
		end_if

		kill_entity own_id
	else_if world_x > RIGHT_SIDE_BASE_X then
		if wave_continue_mode = WAIT_UNTIL_LAST_DEAD then
			let parent.total_alive_children = parent.total_alive_children - 1
		end_if

		kill_entity own_id
	end_if

return

// ---------------- Bullet Firing Functions ----------------
// These functions deal with all aspects of shootin' 'n' shit.

.fire_shots

// This function is called by all enemies and uses the generic properties that they
// all possess in order to either fire a spray of bullets, or to fire at the player.

	if firing_behaviour & BULLET_FREQUENCY_RANDOM then

		LET firing_chance = SPECIAL_RAND (0, firing_frequency)

		IF firing_chance = 0 THEN

			gosub fire_shot

		END_IF

	else_if firing_behaviour & BULLET_FREQUENCY_FIXED then

		let firing_counter = firing_counter - 1 !< 0

		if firing_counter = 0 then

			let firing_counter = firing_frequency

			gosub fire_shot

		end_if

	else

		// Could have no type of timing behaviour if it's a special one or just doesn't fire bullets.

	end_if

return



.fire_shot

	if firing_behaviour & BULLET_TYPE_SINGLE_DIRECTED then
		gosub fire_directed_shot
	else_if firing_behaviour & BULLET_TYPE_SPREAD then
		gosub fire_spread_shot
	else_if firing_behaviour & BULLET_TYPE_CLOCKWISE then
		gosub fire_clockwise_shot
	else
		STOP // Uh-oh!
	end_if

return



.fire_clockwise_shot

	PLAY_SOUND enemy_fire_single_bullet 255 CONST_SOUND_FREQUENCY

	SPAWN_ENTITY (enemy_bullet,0,0,process_next)
	LET LAST_CHILD.movement_angle = clockwise_bullet_angle
	LET clockwise_bullet_angle = clockwise_bullet_angle + 2250
	LET player_on_level_number_var = get_global_flag (player_on_level_number)
	LET bullet_speed_modifier = 10000 / CONST_NUMBER_OF_LEVELS_MINUS_ONE
	LET bullet_speed_modifier = bullet_speed_modifier * player_on_level_number_var
	let bullet_speed_modifier = bullet_speed_modifier % enemy_bullet_speed_percentage
	LET LAST_CHILD.movement_speed = LERP (MINIMUM_ENEMY_BULLET_SPEED, MAXIMUM_ENEMY_BULLET_SPEED, bullet_speed_modifier)

return



.fire_spread_shot

	PLAY_SOUND enemy_fire_bullet_spread 255 CONST_SOUND_FREQUENCY

	FOR TEMP_1 = 0 TO 35999 STEP 9000
		SPAWN_ENTITY (enemy_bullet,0,0,process_next)
		LET LAST_CHILD.movement_angle = TEMP_1
		LET player_on_level_number_var = get_global_flag (player_on_level_number)
		LET bullet_speed_modifier = 10000 / CONST_NUMBER_OF_LEVELS_MINUS_ONE
		LET bullet_speed_modifier = bullet_speed_modifier * player_on_level_number_var
		let bullet_speed_modifier = bullet_speed_modifier % enemy_bullet_speed_percentage
		LET LAST_CHILD.movement_speed = LERP (MINIMUM_ENEMY_BULLET_SPEED, MAXIMUM_ENEMY_BULLET_SPEED, bullet_speed_modifier)
	NEXT

	FOR TEMP_1 = 4500 TO 35999 STEP 9000
		SPAWN_ENTITY (enemy_bullet,0,0,process_next)
		LET LAST_CHILD.movement_angle = TEMP_1
		LET player_on_level_number_var = get_global_flag (player_on_level_number)
		LET bullet_speed_modifier = 10000 / CONST_NUMBER_OF_LEVELS_MINUS_ONE
		LET bullet_speed_modifier = bullet_speed_modifier * player_on_level_number_var
		let bullet_speed_modifier = bullet_speed_modifier % enemy_bullet_speed_percentage
		LET LAST_CHILD.movement_speed = LERP (MINIMUM_ENEMY_BULLET_SPEED, MAXIMUM_ENEMY_BULLET_SPEED, bullet_speed_modifier)
		LET LAST_CHILD.movement_speed = LAST_CHILD.movement_speed % 7071
	NEXT

return



.fire_directed_shot

	PLAY_SOUND enemy_fire_single_bullet 255 CONST_SOUND_FREQUENCY

	LET TEMP_1 = GET_GLOBAL_FLAG (wizball_entity_id_flag)

	IF TEMP_1 != CONST_UNSET then
		SPAWN_ENTITY (enemy_bullet,0,0,process_next)
		LET LAST_CHILD.movement_angle = ANGLE_TO_ENTITY (TEMP_1)
		LET TEMP_1 = SPECIAL_RAND (0, -ENEMY_BULLET_INNACCURACY,ENEMY_BULLET_INNACCURACY)
		LET LAST_CHILD.movement_angle = LAST_CHILD.movement_angle + TEMP_1
		LET player_on_level_number_var = get_global_flag (player_on_level_number)
		LET bullet_speed_modifier = 10000 / CONST_NUMBER_OF_LEVELS_MINUS_ONE
		LET bullet_speed_modifier = bullet_speed_modifier * player_on_level_number_var
		let bullet_speed_modifier = bullet_speed_modifier % enemy_bullet_speed_percentage
		LET LAST_CHILD.movement_speed = LERP (MINIMUM_ENEMY_BULLET_SPEED, MAXIMUM_ENEMY_BULLET_SPEED, bullet_speed_modifier)
	END_IF

return
