// This is the text used for the options. It simply fades in and out when it sees the signal from it's parent. Really fast.

#define current_menu_entity_id			v0
#define current_position_in_name_array		v1

#define current_letter_number			v3
#define current_letter_x			v4
#define current_letter_y			v5
#define current_letter_width			v6
#define current_letter_entity_id		v7

#define current_name_width			v9
#define current_name_base_position		v10

#define frame_width				v11
#define frame_height				v12

#define start_x					v20
#define start_y					v21
#define target_x				v22
#define target_y				v23
#define progress_percentage			v24
#define bias_progress_percentage		v25
#define ideal_target_x				v26
#define ideal_target_y				v27

#define final_phase				v28

#define removed_letter_width			v44

// The current_target_x and current_target_y migrate to the ideal ones
// by a pixel every frame. Really it should only occur on the x-axis, but
// I may as well future proof it.

	let final_phase = false

	let progress_percentage = 0

	let draw_order = MENU_TEXT_DRAW_ORDER

	let entity_type = ENT_TYPE_PERMANENT_MENU_ENTITY

	let draw_mode = DRAW_MODE_SPRITE

	let opengl_booleans = opengl_boolean_blend_add + opengl_boolean_vertex_colour_alpha

	let sprite = font[arb]

	let opengl_vertex_alpha = 255

	let current_frame = current_letter_entity_id.current_frame
	let start_x = current_letter_entity_id.world_x
	let start_y = current_letter_entity_id.world_y

	let frame_width = GET_GRAPHIC_FRAME_WIDTH (sprite,current_frame)
	let frame_height = GET_GRAPHIC_FRAME_HEIGHT (sprite,current_frame)



start

	let progress_percentage = progress_percentage + 200 !> 10000

	let bias_progress_percentage = progress_percentage curve_percentage CURVE_PERCENTAGE_SISO

	let world_x = lerp (start_x, target_x, bias_progress_percentage)
	let world_y = lerp (start_y, target_y, bias_progress_percentage)

	let target_x = parent.current_name_base_position + current_name_width
	let target_y = HISCORE_NAME_CENTER_Y

	set_private_co_ords

	if final_phase = false then
		if parent.current_position_in_name_array <= current_position_in_name_array then
			let final_phase = true
			let parent.removed_letter_width = frame_width
		end_if
	end_if

	let current_menu_entity_id = GET_GLOBAL_FLAG (menu_entity_identity_flag)

	if current_menu_entity_id != CONST_UNSET then
		if current_menu_entity_id.signal & DESTROY_HISCORE_TEXT then
			let final_phase = true
		end_if
	end_if

	if final_phase = true then
		let opengl_vertex_alpha = opengl_vertex_alpha - 15 !< 0

		if opengl_vertex_alpha = 0 then
			kill_entity own_id
		end_if
	end_if

end
