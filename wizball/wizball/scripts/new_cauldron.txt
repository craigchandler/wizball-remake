#define COLOUR_NUMBER			passed_param_1
	// This is so they can read the main_game_controller's array to see how full they should be.

#define INHERIT_COLOUR			passed_param_1
#define DRAW_ORDER_OFFSET		passed_param_2
#define SPECIAL_OPENGL_BOOLEAN		passed_param_3

#define CURRENT_RED			V0
#define CURRENT_GREEN			V1
#define CURRENT_BLUE			V2

#define TARGET_RED			V3
#define TARGET_GREEN			V4
#define TARGET_BLUE			V5

#define COLOUR_CHANGE_PROGRESS		V6

	LET WINDOW_NUMBER = GAME_PANEL_WINDOW
	LET SPRITE = title_screen_and_large_bits[arb]

	LET BASE_FRAME = 18
	LET CURRENT_FRAME = BASE_FRAME

	LET DRAW_MODE = DRAW_MODE_SPRITE
	LET DRAW_ORDER = CAULDRON_DRAW_ORDER

	LET OPENGL_BOOLEANS = OPENGL_BOOLEAN_VERTEX_COLOUR + OPENGL_BOOLEAN_BLEND_ADD

	SWITCH COLOUR_NUMBER

		CASE CAULDRON_RED
			LET OPENGL_VERTEX_RED = 255
			LET OPENGL_VERTEX_GREEN = 0
			LET OPENGL_VERTEX_BLUE = 0

			LET WORLD_X = 448
		BREAK

		CASE CAULDRON_GREEN
			LET OPENGL_VERTEX_RED = 0
			LET OPENGL_VERTEX_GREEN = 255
			LET OPENGL_VERTEX_BLUE = 0

			LET WORLD_X = 496
		BREAK

		CASE CAULDRON_BLUE
			LET OPENGL_VERTEX_RED = 0
			LET OPENGL_VERTEX_GREEN = 0
			LET OPENGL_VERTEX_BLUE = 255

			LET WORLD_X = 544
		BREAK

		CASE CAULDRON_COMBINATION

			// Because the first colour on the first level is red, innit?
			LET OPENGL_VERTEX_RED = 255
			LET OPENGL_VERTEX_GREEN = 0
			LET OPENGL_VERTEX_BLUE = 0

			LET WORLD_X = 608

			LET CURRENT_RED = OPENGL_VERTEX_RED
			LET CURRENT_GREEN = OPENGL_VERTEX_GREEN
			LET CURRENT_BLUE = OPENGL_VERTEX_BLUE
		BREAK

	END_SWITCH

	LET COLOUR_CHANGE_PROGRESS = -9000

	LET WORLD_Y = 32

	let bitshift = 0
	set_private_co_ords

	spawn_entity (new_cauldron_liquid_underlay,0,44,process_next)
	LET last_child.COLOUR_NUMBER = COLOUR_NUMBER

	// Spawn back panel...
	SPAWN_ENTITY (NEW_CAULDRON_LAYER, 0, 0, PROCESS_NEXT)
	LET last_child.INHERIT_COLOUR = TRUE
	LET last_child.DRAW_ORDER_OFFSET = -3
	LET last_child.SPECIAL_OPENGL_BOOLEAN = OPENGL_BOOLEAN_MASKED
	LET last_child.CURRENT_FRAME = 14

	// Spawn liquid negator layer
	SPAWN_ENTITY (NEW_CAULDRON_LAYER, 0, 0, PROCESS_NEXT)
	LET last_child.INHERIT_COLOUR = FALSE
	LET last_child.DRAW_ORDER_OFFSET = -1
	LET last_child.SPECIAL_OPENGL_BOOLEAN = OPENGL_BOOLEAN_BLEND_SUBTRACT
	LET last_child.CURRENT_FRAME = 17

	// Spawn Mask Element
	SPAWN_ENTITY (NEW_CAULDRON_LAYER, 0, 0, PROCESS_NEXT)
	LET last_child.INHERIT_COLOUR = FALSE
	LET last_child.DRAW_ORDER_OFFSET = 1
	LET last_child.SPECIAL_OPENGL_BOOLEAN = OPENGL_BOOLEAN_MASKED
	LET last_child.CURRENT_FRAME = 19

	// Spawn Shine Overlay
	SPAWN_ENTITY (NEW_CAULDRON_LAYER, 0, 0, PROCESS_NEXT)
	LET last_child.INHERIT_COLOUR = FALSE
	LET last_child.DRAW_ORDER_OFFSET = 2
	LET last_child.SPECIAL_OPENGL_BOOLEAN = OPENGL_BOOLEAN_BLEND_ADD
	LET last_child.CURRENT_FRAME = 20





start

	if COLOUR_NUMBER = CAULDRON_COMBINATION THEN

		LET TEMP_1 = GET_GLOBAL_FLAG (MAIN_GAME_CONTROLLER_ENTITY_ID)

		IF TEMP_1 != CONST_UNSET THEN

			LET TARGET_RED = READ_FROM_ARRAY (TEMP_1,COMBO_CAULDRON_RGB_ARRAY_ID,0)
			LET TARGET_GREEN = READ_FROM_ARRAY (TEMP_1,COMBO_CAULDRON_RGB_ARRAY_ID,1)
			LET TARGET_BLUE = READ_FROM_ARRAY (TEMP_1,COMBO_CAULDRON_RGB_ARRAY_ID,2)

			// Man we need multiple arguments in our IF statements badly!

			IF CURRENT_RED != TARGET_RED THEN
				gosub update_colour
			ELSE_IF CURRENT_GREEN != TARGET_GREEN THEN
				gosub update_colour
			ELSE_IF CURRENT_BLUE != TARGET_BLUE THEN
				gosub update_colour
			END_IF

		ELSE

			STOP // WTF?!

		END_IF

	END_IF

end



.update_colour

	// Okay, this changes our colour towards the target colour gradually. When it reaches it, it stops. Dur.

	LET COLOUR_CHANGE_PROGRESS = COLOUR_CHANGE_PROGRESS + 500 !> 9000

	LET TEMP_1 = 5000 SIN COLOUR_CHANGE_PROGRESS
	LET TEMP_1 = TEMP_1 + 5000

	LET OPENGL_VERTEX_RED = LERP (CURRENT_RED, TARGET_RED, TEMP_1)
	LET OPENGL_VERTEX_GREEN = LERP (CURRENT_GREEN, TARGET_GREEN, TEMP_1)
	LET OPENGL_VERTEX_BLUE = LERP (CURRENT_BLUE, TARGET_BLUE, TEMP_1)

	IF COLOUR_CHANGE_PROGRESS = 9000 THEN
		LET COLOUR_CHANGE_PROGRESS = -9000 // In readiness for next time
		LET CURRENT_RED = TARGET_RED
		LET CURRENT_GREEN = TARGET_GREEN
		LET CURRENT_BLUE = TARGET_BLUE
	END_IF

	return
