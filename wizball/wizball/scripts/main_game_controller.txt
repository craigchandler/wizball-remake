// This sits in the background all the time of the game and deals with the spawning of pearls, regular enemy waves, the police, etc.

// How Wizball works:

// When the player starts a new life, all enemies/pearls are destroyed and the
// enemy counts are zeroed. The current level then has pearl molecules spawned
// into it.

// During normal running if the number of enemies on the current level falls to
// to 0 then we spawn a new set of enemies. As this is a constant check we don't
// have to worry about spawning enemies on the other levels until we enter them.

// Each time an enemy is created or destroyed in a level it alters the array in this

CREATE_ARRAY (LEVEL_ENEMY_COUNT_ARRAY_ID,CONST_NUMBER_OF_LEVELS)
// This array knows how many enemies are in each level. Oh yuss.

CREATE_ARRAY (LEVEL_COMPLETION_ARRAY_ID,CONST_NUMBER_OF_LEVELS)
// This array knows how complete each level is. ie, which of its
// colours have been filled in. It's updated by other entities
// which also then fill in the LEVEL_ACCESS_ARRAY. This is then
// used to spawn the player into a new level if the new one is
// all finished.

CREATE_ARRAY (LEVEL_ACCESS_ARRAY_ID,CONST_NUMBER_OF_LEVELS)
// This array is queried when the player is travelling from one
// level to another in order to figure out the destination level.
// The actual position within it is random.

CREATE_ARRAY (END_OF_BONUS_LEVEL_SUMMARY_ARRAY_ID,32)
// This is used to write the text detailing your exploits in the bonus level to.

CREATE_ARRAY (CAULDRON_FULLNESS_ARRAY_ID,4)
// We have 4 slots so that the final cauldron's total contents can be stored here, too, so that all the
// cauldron entities can read their stuff easily.

CREATE_ARRAY (COMBO_CAULDRON_RGB_ARRAY_ID,3)
// This contains the RGB values for the combination cauldron, so that when transfering levels the one on the
// panel can smoothly fade to the new values in a funky manner while emtpying/filling itself to the new
// level.

CREATE_ARRAY (DRAINING_TRACKING_ARRAY_ID,3)
// This array is initialised to the needed amount for the particular colour that's being drained so that we
// can deplete the cauldrons in order nicely.

// This is where the wizball spins off into the unknown for the bonus level, but before the actual bonus level.

// This is where the entity goes on it's holidays when Wizball completes a colour and its
// services aren't needed. It'll be given a kick up the bum when it's time to come out of it.

.bonus_level_handler

	wait 45

	set_global_flag (continue_save_allowed,false)

	gosub stop_music

	wait 15

	gosub start_bonus_level_music

	set_global_flag (bonus_level_enemies_killed,0)

	set_global_flag (currently_on_bonus_level,true)

	alter_entities ( ALTERATION_SLEEP , ENTITY_TYPE & ENT_TYPE_SLEPT_IN_LEVEL_TRANSITION )
	
	// Kill everything that needs to be killed in a level transistion. That being:
	// Wizball, bullets, level masking elements, paintdrops, game window handlers.
	alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_DESTROYED_IN_LEVEL_TRANSITION )
	set_global_flag (current_existant_enemy_bullets, 0)

	// Because we don't want to set up a proper game_window_handler, we just do the essentials ourselves
	// here and spawn summat funky for the background.
	SET_CURRENT_COLLISION_TILEMAP (EMPTY_TILEMAP)
	SETUP_COLLISION_SYSTEM_SIZE_BY_COLLISION_MAP (6)
	let current_zone = get_current_zone (0,0,game_room_zone)
	SETUP_ACTIVE_COLLISION_WINDOW_BY_ZONE (current_zone)
	SET_WINDOW_MAP_POSITION (0, 0)

	// And set up the wave spawning variables...
	let wave_number = -1
	let wave_wait = 50
	let wave_type = CONST_UNSET

	// And pop a reminder to start spawning shit on the queue.
	ADD_EVENT_TO_QUEUE ( OWN_ID , LEVEL_RESET_FLAG_MOVE_TO_NEXT_BONUS_WAVE )

	let total_bonus_wave_aliens_killed = 0

	spawn_entity (bonus_level_plasma_cloud,0,0,process_next)
	spawn_entity (bonus_level_plasma_cloud,640,416,process_next)
	spawn_entity (bonus_level_starfield,0,0,process_next)
	spawn_entity (bonus_level_nebula,0,0,process_next)

	set_global_flag (bonus_level_starfield_slow_flag,0)

	spawn_entity (wizball,320,208,process_next)

	let bonus_level_summary_counter = 0

	BLANK_WINDOW (2)

	let re_enable_pause_wait_counter = 10

start

	let re_enable_pause_wait_counter = re_enable_pause_wait_counter - 1 !< 0
	if re_enable_pause_wait_counter = 1 then
		set_global_flag (pause_allowed, true)
	end_if

	let reset_flag = GET_NEXT_EVENT_FROM_QUEUE

	// Okay, we could easily get a flag arrive at any time, not only when we're ready to spawn a wave.

	if reset_flag = LEVEL_RESET_FLAG_MOVE_TO_NEXT_BONUS_WAVE then
		if (wave_number >= 0) then
			let wave_wait = READ_FROM_DATATABLE (bonus_wave_order, wave_number, 4)
		else
			let wave_wait = 50
		end_if
		let wave_number = wave_number + 1
		let wave_type = READ_FROM_DATATABLE (bonus_wave_order, wave_number, 0)
	else_if reset_flag = LEVEL_RESET_FLAG_GO_TO_LABORATORY then
		goto laboratory_screen
	else_if reset_flag = LEVEL_RESET_FLAG_WIZBALL_FINISHED then
		let bonus_level_summary_counter = 1
	else_if reset_flag = LEVEL_RESET_FLAG_BONUS_ENEMY_KILLED then
		let total_bonus_wave_aliens_killed = total_bonus_wave_aliens_killed + 1
	end_if

	if bonus_level_summary_counter > 0 then

			let text_counter = 0
			reset_array (own_id, END_OF_BONUS_LEVEL_SUMMARY_ARRAY_ID)

			LET text_counter = WRITE_TEXT_TO_ARRAY (own_id, END_OF_BONUS_LEVEL_SUMMARY_ARRAY_ID , BONUS_START_TAG, text_counter, TRUE)
			let temp_1 = get_global_flag (bonus_level_enemies_killed)
			LET text_counter = WRITE_TEXT_NUMBER_TO_ARRAY (own_id, END_OF_BONUS_LEVEL_SUMMARY_ARRAY_ID , temp_1, text_counter, TRUE)
			LET text_counter = WRITE_TEXT_TO_ARRAY (own_id, END_OF_BONUS_LEVEL_SUMMARY_ARRAY_ID , BONUS_END_TAG, text_counter, TRUE)
			LET temp_1 = temp_1 * 40
			LET text_counter = WRITE_TEXT_NUMBER_TO_ARRAY (own_id, END_OF_BONUS_LEVEL_SUMMARY_ARRAY_ID , temp_1, text_counter, FALSE)

			let temp_1 = get_global_flag (bonus_level_enemies_killed)
			let temp_1 = temp_1 * 40

			let temp_2 = get_global_flag (player_score)
			let temp_2 = temp_2 + temp_1 !> MAXIMUM_POSSIBLE_SCORE
			set_global_flag (player_score,temp_2)

		end_if

		let bonus_level_summary_counter = bonus_level_summary_counter + 1

		if bonus_level_summary_counter = 200 then
			ADD_EVENT_TO_QUEUE (own_id , LEVEL_RESET_FLAG_GO_TO_LABORATORY)
		end_if
	end_if

	let wave_wait = wave_wait - 1 !< 0

	if wave_wait = 0 then
		// Okay, there's no after-wave delay to *stop* us from spawning a new wave, but have we
		// received the node from the last wave spawner we created to go ahead yet?

		if wave_type != CONST_UNSET then

			// We spawn the next wave in the list! Woo!

			if wave_type = BONUS_WAVE_FINISHED then
				goto laboratory_screen
			else_if wave_type = BONUS_WAVE_SLOW_STARS then
				// Slow the stars!
				set_global_flag (bonus_level_starfield_slow_flag,1)
				// And automatically move onto the next bit...
				ADD_EVENT_TO_QUEUE (own_id , LEVEL_RESET_FLAG_MOVE_TO_NEXT_BONUS_WAVE)
			else
				// Spawn the wave spawner and pass along all the dirt...

				spawn_entity bonus_wave_spawner,0,0,process_next
				let last_child.passed_param_1 = wave_type
				let last_child.passed_param_2 = READ_FROM_DATATABLE (bonus_wave_order, wave_number, 1)
				let last_child.passed_param_3 = READ_FROM_DATATABLE (bonus_wave_order, wave_number, 2)
				let last_child.passed_param_4 = READ_FROM_DATATABLE (bonus_wave_order, wave_number, 3)
				let last_child.passed_param_5 = wave_number

				let wave_type = CONST_UNSET
			end_if
		end_if
	end_if

end



// This is the laboratory screen which is reached either after completing the bonus level,
// or after popping your clogs on the bonus level. Like I give a shit which. ;)

	// Kill off the enemies and bullets and wizball, if need-be.

	alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_BONUS_LEVEL_ENTITY )
	alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_DESTROYED_IN_LEVEL_TRANSITION )
	set_global_flag (current_existant_enemy_bullets, 0)

	wait 1

	spawn_entity (laboratory_starfield,0,0,process_next)
	spawn_entity (laboratory_nebula,0,0,process_next)

	gosub swap_level_tiles
	gosub set_up_wanted_colour_draining

	gosub stop_music
	gosub start_wizlab_music

	SPAWN_ENTITY (laboratory, 0, 0, process_previous)

	end_if

	let reset_flag = GET_NEXT_EVENT_FROM_QUEUE

	if reset_flag = DEPLETE_CAULDRON_OF_TARGET_COLOUR_AND_UPDATE then
		gosub drain_wanted_colour
	ELSE_IF reset_flag = LEVEL_RESET_FLAG_GO_TO_FLY_THRU then
		LET currently_ignoring_full_combo_cauldron = FALSE
		goto fly_through_sequence
	end_if

end





// This checks the level of completion of the levels and sets the accessability accordingly.
// If the player has just completed a level then there's a fly-through of it, and the player
// is dumped onto the newly unlocked level (if any).

.fly_through_sequence

	FADE_PERSISTANT_CHANNELS

	set_global_flag (pause_allowed, false)
	set_global_flag (continue_save_allowed,false)

	BLANK_WINDOW (2)

	alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_ATTRACT_SEQUENCE_ENTITY )

	wait 1

	let current_level = get_global_flag (player_on_level_number)

	let level_progress = read_from_array ( own_id , LEVEL_COMPLETION_ARRAY_ID , current_level )

	let level_progress = level_progress + 1

	write_to_array ( level_progress , own_id , LEVEL_COMPLETION_ARRAY_ID , current_level )

	if level_progress = 3 then
		gosub find_highest_accessable_level

		// This is set to a really high number so that we're boosted to the highest level!
		set_global_flag (player_warp_direction,CONST_NUMBER_OF_LEVELS)

		spawn_entity (flythru, 0, 0, process_previous)
	else
		// This is set to 0 so we emerge on the same level.
		set_global_flag (player_warp_direction,0)

		goto exit_to_main_level
	end_if

	let re_enable_pause_wait_counter = 10

start

	let re_enable_pause_wait_counter = re_enable_pause_wait_counter - 1 !< 0
	if re_enable_pause_wait_counter = 1 then
		set_global_flag (pause_allowed, true)
	end_if

	let reset_flag = GET_NEXT_EVENT_FROM_QUEUE

	if reset_flag = LEVEL_RESET_FLAG_EXIT_FLY_THRU then
		alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_BONUS_LEVEL_ENTITY )
		alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_DESTROYED_IN_LEVEL_TRANSITION )
		goto check_for_game_completion
	end_if

end


.check_for_game_completion

	set_global_flag (pause_allowed, false)

	let all_levels_complete = true	

	for check_level_number = 0 to 8

		let temp_1 = read_from_array ( own_id , LEVEL_COMPLETION_ARRAY_ID , check_level_number )

		if temp_1 != 3 then
			let all_levels_complete = false
		end_if
	next

	if all_levels_complete = true then
		goto game_completion_effect
	end_if



// This now kills any enemies on levels which are lower than the minimum open one.

	let counter = 0

start

	if counter < min_open_level THEN
		alter_entities ( ALTERATION_KILL , PARENT_LEVEL = counter )
	end_if

	wait 1

	let counter = counter + 1

	if counter > CONST_NUMBER_OF_LEVELS then
		goto exit_to_main_level
	end_if

end

// This just pops out to the main game again, as if the player just warped to a level.

.exit_to_main_level

	alter_entities ( ALTERATION_KILL , ENTITY_TYPE & ENT_TYPE_ATTRACT_SEQUENCE_ENTITY )

	wait 1

	gosub stop_music

	gosub start_in_game_music

	ADD_EVENT_TO_QUEUE ( OWN_ID , LEVEL_RESET_FLAG_PLAYER_WARPED )

	goto restart_main_loop



// Life Loss Sequence

.reset_due_to_life_loss

	FADE_PERSISTANT_CHANNELS

	set_global_flag (pause_allowed, false)
	set_global_flag (continue_save_allowed,false)

	// If we're in a freakout, then we emerge from it due to the death.

	let temp_1 = get_global_flag (freak_out_manager_entity_id)

	if temp_1 != CONST_UNSET then
		ADD_EVENT_TO_QUEUE ( temp_1 , FREAKOUT_EVENT_EXIT )
	end_if

	// And reduce the life-counter...

	let temp_1 = get_global_flag (player_lives)
	let temp_1 = temp_1 - 1

	if temp_1 > -1 then
		set_global_flag (player_lives,temp_1)
	else
		// GAME OVER!
		goto game_over_routine

	next

.check_for_full_colour_complement

	// This looks at the current level and it's level of completeness and then figures out if we've
	// filled up a lovely colour yet.

	LET level_progress = read_from_array ( own_id , LEVEL_COMPLETION_ARRAY_ID , current_level )

	// Only do the action if we haven't already completed the level...

	IF level_progress < 3 THEN

		LET temp_1 = current_level * 3
		// This is because for each level, we have 3 sets if RGB data.

		let temp_1 = temp_1 + level_progress 
		// Now temp_1 equals the line of data in the level_completion_colours datatable.

		let match_count = 0
		// This will store how many cauldrons we've matched or exceeded the needed amount in.

		let total_contributed_amount = 0
		// This will store how full the combo cauldron is. Oh yuss.

		for counter = 0 to 3

			let cauldron_fullness = READ_FROM_ARRAY (OWN_ID,CAULDRON_FULLNESS_ARRAY_ID,counter)
			let needed_cauldron_fullness = READ_FROM_DATATABLE (level_completion_colours,temp_1,counter)

			let contributed_amount = cauldron_fullness !> needed_cauldron_fullness
			let total_contributed_amount = total_contributed_amount + contributed_amount

			if cauldron_fullness >= needed_cauldron_fullness then
				LET match_count = match_count + 1
			end_if

		next

		WRITE_TO_ARRAY (total_contributed_amount,OWN_ID,CAULDRON_FULLNESS_ARRAY_ID,CAULDRON_COMBINATION)

		IF match_count = 3 then
			IF currently_ignoring_full_combo_cauldron = FALSE then
				// We don't want to clog up the queue while we're in the wizlab draining the cauldrons, but we do
				// wanna' calculate their volume.

				// Oh, and filling a cauldron will stop the freakout, too!
				gosub stop_freakout

				ADD_EVENT_TO_QUEUE ( OWN_ID , LEVEL_RESET_FLAG_GET_READY_FOR_BONUS )
			END_IF
		END_IF

	ELSE

		WRITE_TO_ARRAY (0,OWN_ID,CAULDRON_FULLNESS_ARRAY_ID,CAULDRON_COMBINATION)

	END_IF

	gosub read_combo_cauldron_rgb_values

	return
