	LET ENTITY_TYPE = ENT_TYPE_DESTROYED_IN_LEVEL_TRANSITION + ENT_TYPE_DESTROYED_IN_LEVEL_RESET
	LET ENTITY_TYPE = ENTITY_TYPE OR ENT_TYPE_PAUSABLE

// This is an enemy bullet. Yuss.

#define	colour_angle		V0
#define self_destructing	V1
#define movement_angle		V2
#define movement_speed		V3



#define OVERRIDE_SIZE_X		PASSED_PARAM_1
#define OVERRIDE_SIZE_Y		PASSED_PARAM_2

	let temp_1 = get_global_flag (current_existant_enemy_bullets)

	if temp_1 >= MAX_ENEMY_BULLETS then
		kill_entity own_id
	else
		let temp_1 = temp_1 + 1
		set_global_flag (current_existant_enemy_bullets, temp_1)
	end_if

// ---------------- Bitshift And Anchor ---------------- 

	LET BITSHIFT = 8
	SET_PRIVATE_CO_ORDS

// ---------------- Appearance Set Up ---------------- 

	LET OPENGL_BOOLEANS = OPENGL_BOOLEAN_SCALE + OPENGL_BOOLEAN_BLEND_ADD
	LET OPENGL_BOOLEANS = OPENGL_BOOLEANS + OPENGL_BOOLEAN_FILTERED

	LET SPRITE = particle_effects[arb]
	LET BASE_FRAME = 19

	LET CURRENT_FRAME = BASE_FRAME 
	LET FRAME_COUNT = 1

	LET DRAW_ORDER = ENEMY_BULLET_DRAW_ORDER

	LET OPENGL_SCALE_X = 15000
	LET OPENGL_SCALE_Y = 15000
	LET OPENGL_ANGLE = 0

// ---------------- Object And World Collision Set Up ---------------- 

	ENTITY_HITLINE GOSUB any_interaction_routine
	WORLD_HITLINE GOSUB disappear_without_a_trace

	LET COLLIDE_TYPE = ENT_TYPE_ENEMY_BULLET
	LET COLLIDE_WITH = ENT_TYPE_PLAYER + ENT_TYPE_CATELLITE

	LET WORLD_COLLISION_BEHAVIOUR = COLLISION_RESPONSE_SIMPLE_BOUNCE_OR_DECELLERATE + COLLISION_USE_EXTRA_TEST_POINTS
	LET WORLD_COLLISION_BEHAVIOUR = WORLD_COLLISION_BEHAVIOUR + COLLISION_ITERATE_MOVEMENT 
	LET WORLD_COLLISION_BEHAVIOUR = WORLD_COLLISION_BEHAVIOUR + COLLISION_RESPONSE_CALL_SCRIPT 
	let WORLD_COLLISION_BEHAVIOUR = WORLD_COLLISION_BEHAVIOUR + COLLISION_NOTICE_WHEN_INSIDE_COLLISION

	LET ENTITY_MOVEMENT_BEHAVIOUR = ENTITY_MOVEMENT_HORIZONTAL_VERTICAL 

	let WORLD_COLLISION_COEF_HORIZONTAL = 0
	let WORLD_COLLISION_COEF_VERTICAL = 0

	LET WORLD_COLLISION_BITMASK = WORLD_BITMASK_PLAYER_ENEMY_BULLET_COLLIDES
	LET WORLD_COLLISION_LAYER = PLAYER_AND_ENEMY_WORLD_COLLISION_LAYER

	LET COLLISION_SHAPE = RECTANGLE 

	SET_COLLISION_FROM_FRAME -20
	SET_WORLD_COLLISION_FROM_OBJECT

	WAIT 1

	WORLD_HITLINE GOSUB any_interaction_routine

	LET DRAW_MODE = DRAW_MODE_SPRITE

// And... GO!

	LET self_destructing = FALSE

	LET X_VEL = movement_speed SIN movement_angle
	LET Y_VEL = -movement_speed COS movement_angle

start

	call_script function_normal_enemy_am_i_on_screen

	if returned_value_1 = false then
		goto disappear_without_a_trace
	end_if

	IF OPENGL_SCALE_X > 3000 THEN
		LET OPENGL_SCALE_X = OPENGL_SCALE_X >% 3000 2000
		LET OPENGL_SCALE_Y = OPENGL_SCALE_X
	END_IF

	IF self_destructing = TRUE THEN
		LET OPENGL_SCALE_X = OPENGL_SCALE_X - 500 !< 500
		LET OPENGL_SCALE_Y = OPENGL_SCALE_X

		IF OPENGL_SCALE_X = 500 THEN
			gosub remove_from_count
			KILL_ENTITY OWN_ID
		END_IF
	END_IF

end

.any_interaction_routine

	SPAWN_ENTITY (PARTICLE_BLINDER,0,0,PROCESS_NEXT)
	LET last_child.OVERRIDE_SIZE_X = 1500
	LET last_child.OVERRIDE_SIZE_Y = 1500

	PLAY_SOUND enemy_bullet_ping 255 CONST_SOUND_FREQUENCY

	LET self_destructing = TRUE
	LET COLLISION_SHAPE = 0

	LET WORLD_COLLISION_BITMASK = 0

	LET X_VEL = 0
	LET Y_VEL = 0

	end

.disappear_without_a_trace

	gosub remove_from_count

	kill_entity own_id

	end

.remove_from_count

	let temp_1 = get_global_flag (current_existant_enemy_bullets)
	let temp_1 = temp_1 - 1
	set_global_flag (current_existant_enemy_bullets, temp_1)

	return
