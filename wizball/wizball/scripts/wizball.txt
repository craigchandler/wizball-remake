.do_warp_out

	let x_vel = 0
	let y_vel = 0

	IF warp_effect_counter < 30 THEN

		let temp_1 = 10000 / 30
		let temp_1 = temp_1 * warp_effect_counter
		let temp_1 = temp_1 curve_percentage CURVE_PERCENTAGE_SISO
		let world_x = LERP (warp_start_world_x,warp_tube_id.world_x,temp_1)
		let world_y = LERP (warp_start_world_y,warp_tube_id.world_y,temp_1)

		set_private_co_ords

	ELSE_IF warp_effect_counter = 50 THEN

		spawn_entity (particle_blinder,0,0,process_next)

	ELSE_IF warp_effect_counter = 52 THEN

		LET draw_mode = draw_mode_invisible

		set_global_flag (player_warping_destroy_shield,TRUE)

		let particle_count = get_global_flag (particle_commonness_multiplier)
		let particle_count = 10 * particle_count

		for counter = 1 to particle_count
			let temp_1 = rand (-24,24)
			let temp_2 = rand (-24,24)
			spawn_entity (particle_twinkle,temp_1,temp_2,process_next)
		next

	ELSE_IF warp_effect_counter = 120 THEN

		set_global_flag (player_warping_destroy_shield,FALSE)

		IF warp_tube_id.script_number = warp_zone_down THEN
			set_global_flag (player_warp_direction,-1)
		ELSE_IF warp_tube_id.script_number = warp_zone_up THEN
			set_global_flag (player_warp_direction,1)
		ELSE
			STOP // WTF!?
		END_IF

		let temp_1 = get_global_flag (MAIN_GAME_CONTROLLER_ENTITY_ID)
		ADD_EVENT_TO_QUEUE ( temp_1 , LEVEL_RESET_FLAG_PLAYER_WARPED )

	END_IF

	let warp_spin_speed = warp_effect_counter / 6

	let current_frame = current_frame + warp_spin_speed !< 0 !> 63

	let warp_effect_counter = warp_effect_counter + 1

return

// ---------------- Movement ----------------

.movement_routine

	if movement_style = full_controlled_movement then
		if x_vel < 0 then
			let last_movement_direction = -1
		else_if x_vel > 0 then
			let last_movement_direction = 1
		end_if
	else
		if ideal_x_vel < 0 then
			let last_movement_direction = -1
		else_if ideal_x_vel > 0 then
			let last_movement_direction = 1
		end_if
	end_if

	// Simpler to assume one option...
	let allow_movement = true

	IF weapon_collection & CATELLITE_BITFLAG then
		// If we're in one player...
		IF_FLAG (game_mode_single_player = TRUE) then
			let cat_mutant_state = get_global_flag (mutant_cat_flag)

			// Cat ain't mad...
			if cat_mutant_state = 0 then
				IF_INPUT_PLAYER_CONTROL_REPEAT (PLAYER_1,FIRE_1,catellite_control_threshold,1) then
					let allow_movement = false
				END_IF
			end_if
		END_IF
	end_if

	switch movement_style

		case basic_bounce_movement

			let spin_angle = spin_angle + ideal_x_vel !< 0 !> top_spin_angle
			let current_frame = spin_angle / spin_angle_to_frame_divider !< 0 !> 63

			if allow_movement = true then

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 RIGHT then
					let ideal_x_vel = ideal_x_vel + WIZBALL_X_RESPONSIVENESS !> top_private_vel
				END_IF

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 LEFT then
					let ideal_x_vel = ideal_x_vel - WIZBALL_X_RESPONSIVENESS !< -top_private_vel
				END_IF

			end_if

		break

		case controlled_bounce_movement

			let spin_angle = spin_angle + ideal_x_vel !< 0 !> top_spin_angle
			let current_frame = spin_angle / spin_angle_to_frame_divider !< 0 !> 63

			if allow_movement = true then

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 RIGHT then
					let ideal_x_vel = ideal_x_vel + WIZBALL_X_RESPONSIVENESS !> top_private_vel
				END_IF

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 LEFT then
					let ideal_x_vel = ideal_x_vel - WIZBALL_X_RESPONSIVENESS !< -top_private_vel
				END_IF

				let x_vel = ideal_x_vel

			end_if

		break

		case full_controlled_movement

			let spin_angle = spin_angle + x_vel !< 0 !> top_spin_angle
			let current_frame = spin_angle / spin_angle_to_frame_divider !< 0 !> 63

			if allow_movement = true then

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 RIGHT then
					let x_vel = x_vel + WIZBALL_X_RESPONSIVENESS !> top_private_vel
				else
					IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 LEFT then
						let x_vel = x_vel
					else
						if x_vel > 0 then
							let x_vel = x_vel - WIZBALL_X_DAMPING !< 0
						end_if
					end_if
				END_IF

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 LEFT then
					let x_vel = x_vel - WIZBALL_X_RESPONSIVENESS !< -top_private_vel
				else
					IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 RIGHT then
						let x_vel = x_vel
					else
						if x_vel < 0 then
							let x_vel = x_vel + WIZBALL_X_DAMPING !> 0
						end_if
					end_if
				END_IF

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 DOWN then
					let y_vel = y_vel + WIZBALL_Y_RESPONSIVENESS !> top_private_vel
				else
					if y_vel > 0 then
						let y_vel = y_vel - WIZBALL_Y_DAMPING !< 0
					end_if
				END_IF

				IF_INPUT_PLAYER_CONTROL_DOWN PLAYER_1 UP then
					let y_vel = y_vel - WIZBALL_Y_RESPONSIVENESS !< -top_private_vel
				else
					if y_vel < 0 then
						let y_vel = y_vel + WIZBALL_Y_DAMPING !> 0
					end_if
				END_IF

			else

				if x_vel < 0 then
					let x_vel = x_vel + WIZBALL_X_DAMPING !> 0
				else_if x_vel > 0 then
					let x_vel = x_vel - WIZBALL_X_DAMPING !< 0
				end_if

				if y_vel < 0 then
					let y_vel = y_vel + WIZBALL_Y_DAMPING !> 0
				else_if y_vel > 0 then
					let y_vel = y_vel - WIZBALL_Y_DAMPING !< 0
				end_if

			end_if

		break

	end_switch


return

	end_if

.hit_floor_or_roof

	let temp_1 = ABS (y_vel)
	let temp_2 = UNLERP (0, top_private_vel, temp_1)
	let temp_1 = 96 % temp_2 !> 96

	PLAY_SOUND wizball_bounce temp_1 CONST_SOUND_FREQUENCY

	if movement_style != full_controlled_movement then

		let x_vel = ideal_x_vel

		let temp_1 = sgn y_vel // This is the direction we'll be moving off in, but how fast? HOW FAST?!

		// Use the old s = (a * t * t) / 2 formula to find "t".
		// t = sqrt ( (2 * s) / a)

		let temp_2 = WIZBALL_START_Y << bitshift
		let temp_2 = y - temp_2 // temp_2 now equals "s" in the formula. ie, the distance travelled.
		let temp_2 = temp_2 * 2
		let temp_2 = temp_2 / y_acc
		let temp_2 = sqr temp_2 // Okay, now we have "t"

		// So we use the old velocity = accelleration * time

		let y_vel = y_acc * temp_2
		let y_vel = y_vel * temp_1

	else

		// Just let it bounce... But top up the speed to at least a certain level so we don't get "grinding"

		let temp_1 = abs y_vel

		if temp_1 < WIZBALL_MINIMUM_VERTICAL_BOUNCE_SPEED then
			let temp_1 = sgn (y_vel)
			let y_vel = temp_1 * WIZBALL_MINIMUM_VERTICAL_BOUNCE_SPEED 
		end_if

	end_if

//	let temp_1 = get_global_flag (freak_out_manager_entity_id)
//	if temp_1 != CONST_UNSET then
//		ADD_EVENT_TO_QUEUE ( temp_1 , FREAKOUT_EVENT_SMALL_BUMP )
//	end_if

return

.hit_side
	// This is called after the main program itself has bounced the value of x_vel.

	let temp_1 = ABS (x_vel)
	let temp_2 = UNLERP (0, top_private_vel, temp_1)
	let temp_1 = 96 % temp_2 !> 96

	PLAY_SOUND wizball_bounce temp_1 CONST_SOUND_FREQUENCY

	if movement_style = basic_bounce_movement then

		let temp_1 = ideal_x_vel * x_vel

		if temp_1 < 0 then // The two velocities were opposite signs, so we need to fully reflect the x velocity.
			let x_vel = -ideal_x_vel
			let ideal_x_vel = x_vel
		else // The player was spinning Wizball in the direction in readiness for the hit, so we can just use the new ideal velocity.
			let x_vel = ideal_x_vel
		end_if


	else

		let temp_1 = abs x_vel

		if temp_1 < WIZBALL_MINIMUM_HORIZONTAL_BOUNCE_SPEED then
			let temp_1 = sgn (x_vel)
			let x_vel = temp_1 * WIZBALL_MINIMUM_HORIZONTAL_BOUNCE_SPEED
		end_if

		let ideal_x_vel = x_vel

	end_if

	let temp_1 = get_global_flag (freak_out_manager_entity_id)
	if temp_1 != CONST_UNSET then
		ADD_EVENT_TO_QUEUE ( temp_1 , FREAKOUT_EVENT_LARGE_BUMP )
	end_if

return

// ---------------- Update Wobble Status ----------------

.update_wobble

	IF_INPUT_PLAYER_CONTROL_HIT (PLAYER_1,FIRE_2) then
		gosub select_icon
		let wobble_reset_countdown = 0
	END_IF

	if allow_movement = true then

		IF_INPUT_PLAYER_CONTROL_HIT (PLAYER_1,LEFT) then

			if wobble_next_direction = next_wobble_either then
				let wobble_next_direction = next_wobble_right
				let wobble_counter = wobble_counter + 1
				let wobble_reset_countdown = WIZBALL_WOBBLE_DELAY
			else_if wobble_next_direction = next_wobble_left then
				let wobble_next_direction = next_wobble_right
				let wobble_counter = wobble_counter + 1
				let wobble_reset_countdown = WIZBALL_WOBBLE_DELAY
			else
				let wobble_counter = 0
				let wobble_reset_countdown = 0
			end_if

		end_if

		IF_INPUT_PLAYER_CONTROL_HIT (PLAYER_1,RIGHT) then

			if wobble_next_direction = next_wobble_either then
				let wobble_next_direction = next_wobble_left
				let wobble_counter = wobble_counter + 1
				let wobble_reset_countdown = WIZBALL_WOBBLE_DELAY
			else_if wobble_next_direction = next_wobble_right then
				let wobble_next_direction = next_wobble_left
				let wobble_counter = wobble_counter + 1
				let wobble_reset_countdown = WIZBALL_WOBBLE_DELAY
			else
				let wobble_counter = 0
				let wobble_reset_countdown = 0
			end_if

		end_if

		if wobble_counter = WIZBALL_BONUS_SELECTION_WOBBLE_THRESHOLD then
			gosub select_icon
			let wobble_reset_countdown = 0
			// Falls through to next bit to reset counters...
		end_if

	end_if

	let wobble_reset_countdown = wobble_reset_countdown - 1 !< 0

	if wobble_reset_countdown = 0 then
		let wobble_counter = 0
		let wobble_next_direction = next_wobble_either
	end_if

return

.pre_equip_wizball

	gosub set_up_movement

	if weapon_collection & INVULNERABILITY_BITFLAG then

		if parent.script_number = wizball_new_life_appear then
			// It's a new life! Woo! so we're toppin gup the shield for sure!

			spawn_entity (wizball_shield_bubble_layer, 0, 0, process_next)
			set_global_flag (wizball_shield_stored_health, SHIELD_STARTING_ENERGY)

			// Check we have a catellite before spawning a shield for him...

			IF weapon_collection & CATELLITE_BITFLAG then
				spawn_entity (catellite_shield_swirl_layer, 0, 0, process_next)
				set_global_flag (cat_shield_stored_health, SHIELD_STARTING_ENERGY)

				let weapon_collection = weapon_collection OR CATELLITE_INVULNERABILITY_BITFLAG
				set_global_flag (wizball_current_loadout , weapon_collection)
			END_IF

		else
			spawn_entity (wizball_shield_bubble_layer, 0, 0, process_next)

			// Check we have a catellite before spawning a shield for him...

			IF weapon_collection & CATELLITE_BITFLAG then
				// Check he should have a shield (as his bitflag is independant of the player's)

				if weapon_collection & CATELLITE_INVULNERABILITY_BITFLAG then
					spawn_entity (catellite_shield_swirl_layer, 0, 0, process_next)
				end_if
			END_IF
		end_if

	end_if

	// Spawn Catellite afterwards (but before) so catellite shield doesn't bork out.
	if weapon_collection & CATELLITE_BITFLAG then
		if_flag (game_mode_single_player = TRUE) then
			spawn_entity (catellite,0,0,process_next)
		else
			spawn_entity (catellite_two_player,0,0,process_next)
		end_if

		if parent.script_number = wizball_new_life_appear then
			set_global_flag (catellite_stored_health,CATELLITE_STARTING_ENERGY)

			// Reset the mutant flag so he isn't born mad.
			set_global_flag (mutant_cat_flag,0)
		end_if

	end_if

	// Hmm, not much to do at all. Cripes.

return

// ---------------- Icon Selection ----------------

.select_icon

	let temp_1 = false

	switch current_pickup_count

		case 1 // Horizontal and Vertical Control - wooters!
			
			if weapon_collection !& LATERAL_CONTROL_BITFLAG then
				let weapon_collection = weapon_collection or LATERAL_CONTROL_BITFLAG
				gosub set_up_movement
				let temp_1 = true
			else_if weapon_collection !& VERTICAL_CONTROL_BITFLAG then
				let weapon_collection = weapon_collection or VERTICAL_CONTROL_BITFLAG
				gosub set_up_movement
				let temp_1 = true
			end_if

		break

		case 2 // Shield and Reverse Fire

			if weapon_collection !& SHIELD_FIRE_BITFLAG then
				let weapon_collection = weapon_collection or SHIELD_FIRE_BITFLAG
				let temp_1 = true
			else_if weapon_collection !& REAR_FIRE_BITFLAG then
				let weapon_collection = weapon_collection or REAR_FIRE_BITFLAG
				let temp_1 = true
			end_if

		break

		case 3 // Catellites

			if weapon_collection !& CATELLITE_BITFLAG then

				// Just about spawned a Catellite, spawn a shield which he'll be immediately processed before.
				if weapon_collection & INVULNERABILITY_BITFLAG then
					spawn_entity (catellite_shield_swirl_layer, 0, 0, process_next)
				end_if

				let weapon_collection = weapon_collection or CATELLITE_BITFLAG
				let temp_1 = true

				if_flag (game_mode_single_player = TRUE) then
					spawn_entity (catellite,0,0,process_next)
				else
					spawn_entity (catellite_two_player,0,0,process_next)
				end_if

				set_global_flag (catellite_stored_health,CATELLITE_STARTING_ENERGY)

				// Reset the mutant flag so he isn't born mad.
				set_global_flag (mutant_cat_flag,0)

			end_if

		break

		case 4 // Double Fire

			if weapon_collection !& DOUBLE_FIRE_BITFLAG then
				let weapon_collection = weapon_collection or DOUBLE_FIRE_BITFLAG
				let temp_1 = true
			end_if

		break

		case 5 // Spray

			if weapon_collection !& WIZ_SPREAD_FIRE_BITFLAG then
				let weapon_collection = weapon_collection or WIZ_SPREAD_FIRE_BITFLAG
				let weapon_collection = weapon_collection AND !CAT_SPREAD_FIRE_BITFLAG
			else
				let weapon_collection = weapon_collection or CAT_SPREAD_FIRE_BITFLAG
				let weapon_collection = weapon_collection AND !WIZ_SPREAD_FIRE_BITFLAG
			end_if

			let temp_1 = true

		break

		case 6 // Smart Bomb

			PLAY_SOUND smart_bomb 255 CONST_SOUND_FREQUENCY 128

			spawn_entity (smart_bomb_shockwave,0,0,process_next)
			let last_child.x_vel = -8
			spawn_entity (smart_bomb_shockwave,0,0,process_next)
			let last_child.x_vel = 8

			let temp_1 = true

		break

		case 7 // Invulnerability!

			if weapon_collection !& INVULNERABILITY_BITFLAG then
				// Okay, Wizball doesn't have shield on...

				let weapon_collection = weapon_collection or INVULNERABILITY_BITFLAG
				spawn_entity (wizball_shield_bubble_layer, 0, 0, process_next)
				set_global_flag (wizball_shield_stored_health, SHIELD_STARTING_ENERGY)

				IF weapon_collection & CATELLITE_BITFLAG then
					if weapon_collection !& CATELLITE_INVULNERABILITY_BITFLAG then
						// Catellite doesn't have a shield. Aw... Give him a shield and update the weapon collection so it knows.
						let weapon_collection = weapon_collection or CATELLITE_INVULNERABILITY_BITFLAG
						set_global_flag (wizball_current_loadout , weapon_collection)

						spawn_entity (catellite_shield_swirl_layer, 0, 0, process_next)
						set_global_flag (cat_shield_stored_health, SHIELD_STARTING_ENERGY)
					else
						// Catellite does have a shield! Woo!
						set_global_flag (cat_shield_stored_health, SHIELD_STARTING_ENERGY)
					end_if
				END_IF

			else
				// If wizzy has a shield, then we just top up it's energy...

				set_global_flag (wizball_shield_stored_health, SHIELD_STARTING_ENERGY)

				IF weapon_collection & CATELLITE_BITFLAG then
					if weapon_collection !& CATELLITE_INVULNERABILITY_BITFLAG then
						// Catellite doesn't have a shield. Aw... Give him a shield and update the weapon collection so it knows.
						let weapon_collection = weapon_collection or CATELLITE_INVULNERABILITY_BITFLAG
						set_global_flag (wizball_current_loadout , weapon_collection)

						spawn_entity (catellite_shield_swirl_layer, 0, 0, process_next)
						set_global_flag (cat_shield_stored_health, SHIELD_STARTING_ENERGY)
					else
						// Catellite does have a shield! Woo!
						set_global_flag (cat_shield_stored_health, SHIELD_STARTING_ENERGY)
					end_if
				END_IF
			end_if

			let temp_1 = true
		break

	end_switch

	if temp_1 = true then
		let current_pickup_count = 0
		PLAY_SOUND bonus_selection 255 CONST_SOUND_FREQUENCY
		set_global_flag (player_bonus_pearl_count,current_pickup_count)
	end_if

	set_global_flag (wizball_current_loadout , weapon_collection)

return
