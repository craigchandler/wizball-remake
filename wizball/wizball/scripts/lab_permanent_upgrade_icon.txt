#define RING_MIDDLE_WORLD_Y			240
#define RING_SCREEN_HALF_HEIGHT			32

		let y_percentage = cos own_angle

		let world_x = RING_SCREEN_HALF_WIDTH % x_percentage
		let world_x = world_x % parent.biased_effect_fullness
		let world_x = world_x + RING_MIDDLE_WORLD_X

		if opengl_scale_x = 0 then
			let opengl_scale_x = 10
		end_if

		let opengl_scale_y = opengl_scale_x

		if y_percentage < 0 then
			let DRAW_ORDER = LABORATORY_CAULDRON_GLOW_DRAW_ORDER - 1

		end_if
