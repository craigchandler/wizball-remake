
#define wiz_id				v0
#define left_of_window			v1
#define map_width			v2
#define map_width_minus_window		v3

#define freakout_flag 			V12
#define freak_out_bump_counter 		V13
#define freak_out_effect_counter 	V14

// Possible Parent defines
#define current_speed			v5

	LET ENTITY_TYPE = ENT_TYPE_DESTROYED_IN_LEVEL_TRANSITION + ENT_TYPE_BONUS_LEVEL_ENTITY
	LET ENTITY_TYPE = ENTITY_TYPE OR ENT_TYPE_PAUSABLE

	let opengl_booleans = opengl_booleans + opengl_boolean_vertex_colour

	let unique_id = MAIN_GAME_STARFIELD_ID
	let draw_mode = DRAW_MODE_STARFIELD

	let draw_override = DRAW_OVERRIDE_OVERRIDE_ALWAYS

	let draw_order = STARFIELD_DRAW_ORDER

start

	let freakout_flag = get_global_flag (freak_out_flag)

	if freakout_flag = TRUE then
		let freak_out_effect_counter = get_global_flag (freak_out_counter)

		if freak_out_effect_counter < FREAKOUT_HALF_EFFECT_LEVEL then
			let draw_mode = DRAW_MODE_STARFIELD
			let draw_override = DRAW_OVERRIDE_OVERRIDE_ALWAYS

			let opengl_vertex_red = freak_out_effect_counter * 255
			let opengl_vertex_red = opengl_vertex_red / FREAKOUT_HALF_EFFECT_LEVEL
			let opengl_vertex_red = 255 - opengl_vertex_red
			let opengl_vertex_green = opengl_vertex_red 
			let opengl_vertex_blue = opengl_vertex_red 
		else
			let draw_mode = DRAW_MODE_INVISIBLE
			let draw_override = DRAW_OVERRIDE_NONE
		end_if

	else
		let opengl_vertex_red = 255
		let opengl_vertex_green = 255
		let opengl_vertex_blue = 255
		let draw_mode = DRAW_MODE_STARFIELD
		let draw_override = DRAW_OVERRIDE_OVERRIDE_ALWAYS
	end_if

	let wiz_id = get_global_flag (wizball_entity_id_flag)

	if wiz_id != const_unset then
		let temp_1 = wiz_id.x_vel * -25
		update_starfield MAIN_GAME_STARFIELD_ID temp_1 5000 TRUE
	else
		if parent.script_number = flythru then
			let temp_1 = parent.current_speed * 8
			update_starfield MAIN_GAME_STARFIELD_ID temp_1 5000 TRUE
		else
			update_starfield MAIN_GAME_STARFIELD_ID 0 5000 TRUE
		end_if
	end_if

end
